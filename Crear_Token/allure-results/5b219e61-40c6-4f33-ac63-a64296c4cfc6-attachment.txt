pm.test("Validar versión de protocolo TLS incorrecta", function () {
    // Verificar si la respuesta contiene un error
    if (pm.response === undefined || pm.response.error) {
        const errorMessage = pm.response.error || '';
        // Validar que el error contiene la parte específica del mensaje
        if (errorMessage.includes('SSL routines:ssl3_read_bytes:tlsv1 alert protocol')) {
            pm.expect(true).to.be.true;  // Pasar el test si el error es el esperado
        } else {
            pm.expect.fail("El mensaje de error no coincide con el esperado.");
        }
    } else {
        // Si la solicitud tuvo éxito, fallar el test, ya que se esperaba un error
        pm.expect.fail("Se esperaba un error de protocolo TLS y no ocurrió.");
    }
});
